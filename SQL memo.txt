-- データベースの作成（DB名: craft_beer_forecast）
CREATE DATABASE craft_beer_forecast;

-- ↑のDBに接続（psqlなどでは \c craft_beer_forecast を実行）
-- ここから craft_beer_forecast DB内で実行

-- 1. users（ユーザー）テーブル
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR NOT NULL,
    password VARCHAR NOT NULL,
    admin BOOLEAN DEFAULT FALSE,
    delete BOOLEAN DEFAULT FALSE
);

-- 2. weather（天気情報）テーブル
CREATE TABLE weather (
    date DATE PRIMARY KEY,
    temp_max FLOAT,
    temp_min FLOAT,
    weather_main VARCHAR,
    humidity INT,
    wind_speed FLOAT
);

-- 3. beers（ビール）テーブル
CREATE TABLE beers (
    id SERIAL PRIMARY KEY,
    name VARCHAR NOT NULL,
    price INT NOT NULL,
    delete BOOLEAN DEFAULT FALSE
);

-- 4. beer_sales（ビール別販売明細）テーブル
CREATE TABLE beer_sales (
    date DATE,
    beer_id INT,
    quantity INT NOT NULL,
    enter_user_id INT,
    PRIMARY KEY (date, beer_id),
    FOREIGN KEY (beer_id) REFERENCES beers(id),
    FOREIGN KEY (enter_user_id) REFERENCES users(id)
);
